p = int(input())
string = str(input())
l = len(string)
maxim = 0
for t in range (1, p+1):
    for i in range (0, l-t):
        if string[i] != string[i+t]:
            if (i+t) > maxim:
                maxim = i+t    
            break             
if maxim == p-1:
    print(0)
else:    
    print(l-maxim)
